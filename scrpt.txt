NA_TOKENS = {"", "none", "null", "nan", "not available", "error", "na", "n/a"}

def is_missing(v):
    if v is None:
        return True
    try:
        if pd.isna(v):
            return True
    except Exception:
        pass
    return str(v).strip().lower() in NA_TOKENS


def split_multi(v, sep=r"[\^,;/|]"):
    if is_missing(v):
        return []
    return [x.strip() for x in re.split(sep, str(v)) if x.strip()]


def last10_set(v):
    """Return set of last-10-digit phone numbers."""
    nums = set()
    for part in split_multi(v):
        digits = re.sub(r"\D", "", part)
        if len(digits) >= 10:
            nums.add(digits[-10:])
    return nums


def seq_ratio(a, b):
    if not a or not b:
        return 0.0
    return SequenceMatcher(None, a, b).ratio()


def edit_distance(a, b):
    if not a or not b:
        return max(len(a or ""), len(b or ""))
    return Levenshtein.distance(a, b)


def fuzzy_match(a, b, ratio_threshold, max_edits):
    if not a or not b:
        return False
    if a == b:
        return True
    if seq_ratio(a, b) >= ratio_threshold:
        return True
    if edit_distance(a, b) <= max_edits:
        return True
    return False


INVALID_VALUES = {
    "", "null", "none", "nan",
    "not available", "error",
    "na", "n/a", "not found",
    "blank", "unknown", "missing",
    "not applicable",
    "-", "--", "n.a", "n.a."
}

# Substrings that should trigger invalid detection anywhere in the string
INVALID_SUBSTRINGS = (
    "not available",
    "not found",
    "not applicable",
    "n/a",
)

def is_invalid(v):
    """
    Returns True if the value is None, NaN, empty, or contains any known
    NA-like token (case-insensitive, whitespace-tolerant).
    """
    if v is None:
        return True
    try:
        if pd.isna(v):
            return True
    except Exception:
        pass
    s = str(v).strip().lower()
    if s == "" or s in INVALID_VALUES:
        return True
    # Substring safety net (handles "Not Available", "  not available  ", etc.)
    for token in INVALID_SUBSTRINGS:
        if token in s:
            return True
    return False


def find_matching_phone(unst_set, base_set, t):
    """
    Return the actual matched phone number (from base_set) if any pair matches.
    Prefers exact match first, then fuzzy match.
    """
    if not unst_set or not base_set:
        return None
    # Exact match first
    common = unst_set & base_set
    if common:
        return next(iter(common))
    # Fuzzy match
    for x in unst_set:
        for y in base_set:
            if fuzzy_match(x, y, t["ratio"], t["max_edits"]):
                return y
    return None


def check_phone_against_many(unst_val, base_vals, t=THRESHOLDS["phone"]):
    """
    Returns:
        match_flag_str, matched_phone, matched_column

    match_flag_str:
        "yes" = Matched with at least one base column
                OR observed value is 'Not available' / invalid (skip check -> yes)
        "no"  = Not matched
    """
    # If OBSERVED (invoice) value is invalid / not available -> treat as MATCH
    if is_invalid(unst_val):
        return "yes", "", ""

    unst_set = last10_set(unst_val)
    if not unst_set:
        return "yes", "", ""

    found_valid_base = False

    for col_name, val in base_vals.items():
        if is_invalid(val):
            continue

        base_set = last10_set(val)
        if not base_set:
            continue

        found_valid_base = True

        matched_phone = find_matching_phone(unst_set, base_set, t)
        if matched_phone:
            return "yes", matched_phone, col_name

    # If no valid base numbers to compare against, treat as match (skip)
    if not found_valid_base:
        return "yes", "", ""

    # Valid numbers were compared but no match found
    return "no", "", ""

def build_reason(match_flag, matched_phone, matched_col):

                    f"'{matched_col}'")

def split_multi(v, sep=r"[\^,;/|]"):
    if is_missing(v):
        return []
    return [x.strip() for x in re.split(sep, str(v)) if x.strip()]


def last10_set(v):
    """Return set of last-10-digit phone numbers."""
    nums = set()
    for part in split_multi(v):
        digits = re.sub(r"\D", "", part)
        if len(digits) >= 10:
            nums.add(digits[-10:])
    return nums


def seq_ratio(a, b):

def fuzzy_match(a, b, ratio_threshold, max_edits):
    if not a or not b:
        return False
    if a == b:
        return True
    if seq_ratio(a, b) >= ratio_threshold:
        return True
    if edit_distance(a, b) <= max_edits:
        return True
    return False

def find_matching_phone(unst_set, base_set, t):
    """
    Return the actual matched phone number (from base_set) if any pair matches.
    Prefers exact match first, then fuzzy match.
    """
    if not unst_set or not base_set:
        return None
    # Exact match first
    common = unst_set & base_set
    if common:
        return next(iter(common))
    # Fuzzy match
    for x in unst_set:
        for y in base_set:
            if fuzzy_match(x, y, t["ratio"], t["max_edits"]):
                return y
    return None


def check_phone_against_many(unst_val, base_vals, t=THRESHOLDS["phone"]):
    """
    Returns:
        match_flag_str, matched_phone, matched_column

    match_flag_str:
        "yes" = Matched with at least one base column
        "no"  = Not matched (or comparable data missing -> treated as no match)
    """
    if is_invalid(unst_val):
        return "no", "", ""

    unst_set = last10_set(unst_val)
    if not unst_set:
        return "no", "", ""

    found_valid_base = False

    for col_name, val in base_vals.items():
        if is_invalid(val):
            continue

        base_set = last10_set(val)
        if not base_set:
            continue

        found_valid_base = True

        matched_phone = find_matching_phone(unst_set, base_set, t)
        if matched_phone:
            return "yes", matched_phone, col_name

    # If no valid base numbers to compare against, treat as no match
    if not found_valid_base:
        return "no", "", ""

    # Valid numbers were compared but no match found
    return "no", "", ""

def check_dealer_name(a, b, t=THRESHOLDS["dealer_name"]):
    """
    a = dealer_name (expected / p_base_data)
    b = dealer_name_unst_str (observed / invoice OCR)

    Returns:
        "yes" = matching (or EITHER side is 'not available' / invalid -> treat as match)
        "no"  = not matching
    """

    # If EITHER value is invalid / not available -> treat as MATCH
    if is_invalid(a) or is_invalid(b):
        return "yes"

    a_norm, b_norm = norm_dealer_name(a), norm_dealer_name(b)

    # If either becomes empty after normalization -> treat as match
    if not a_norm or not b_norm:
        return "yes"

    score = fuzz.token_set_ratio(a_norm, b_norm)  # 0-100

    if score >= t["ratio"] * 100:
        return "yes"

    if edit_distance(a_norm, b_norm) <= t["max_edits"]:
        return "yes"

    return "no"

    a_norm, b_norm = norm_alnum_upper(a), norm_alnum_upper(b)

    # If either becomes empty after normalization -> treat as match
    if not a_norm or not b_norm:
        return "yes"

    if a_norm == b_norm:
        return "yes"

    if len(a_norm) < 12 or len(b_norm) < 12:
        return "no"

    if fuzzy_match(a_norm, b_norm, t["ratio"], t["max_edits"]):

    return "no"

def check_dealer_name(a, b, t=THRESHOLDS["dealer_name"]):
    """
    a = dealer_name (expected / p_base_data)
    b = dealer_name_unst_str (observed / invoice OCR)

    Returns:
        "yes" = matching (or observed value is 'not available' -> treat as match)
        "no"  = not matching

        return "no"

    score = fuzz.token_set_ratio(a_norm, b_norm)  # 0-100

    if score >= t["ratio"] * 100:
        return "yes"

    if edit_distance(a_norm, b_norm) <= t["max_edits"]:

    return "no"

        return []

def clean_phone_set(v):
    """
    Normal cleaning of phone numbers:
      - split on ^ , ; / |
      - remove ALL non-digit characters (spaces, +, -, (), etc.)
      - keep only last 10 digits
      - drop entries that don't have at least 10 digits
    Returns a set of cleaned 10-digit phone numbers.
    """
    nums = set()
    for part in split_multi(v):
        digits = re.sub(r"\D", "", part)   # keep digits only
        if len(digits) >= 10:
            nums.add(digits[-10:])
    return nums

def check_customer_phones_vs_tech(customer_vals):
    """
    customer_vals: dict {col_name: value}
    Returns:
      flag (0/1)         -> 1 if ANY cleaned customer phone exactly matches ANY cleaned tech phone
      observed_values    -> "9876543210 (from customer_contact)"
      matched_tech_info  -> "Tech ID: T123 | Name: Ramesh | Phone: 9876543210"
      matched_cust_cols  -> "customer_contact"
    """
    hits = []  # (cust_phone, cust_col, tech_id, tech_name)

    for col_name, val in customer_vals.items():
        cust_phones = clean_phone_set(val)
        if not cust_phones:
            continue

        # Exact intersection only
        exact_matches = cust_phones & tech_phone_set
        for p in exact_matches:
            for tid, tname in tech_phone_lookup.get(p, []):
                hits.append((p, col_name, tid, tname))

    if not hits:
        return 0, "", "", ""

    # Deduplicate hits
    seen = set()
    uniq = []
    for h in hits:
        key = (h[0], h[1], h[2])
        if key not in seen:
            seen.add(key)
            uniq.append(h)

    observed_values   = "; ".join([f"{h[0]} (from {h[1]})" for h in uniq])
    matched_tech_info = "; ".join([f"Tech ID: {h[2]} | Name: {h[3]} | Phone: {h[0]}" for h in uniq])
    matched_cust_cols = ", ".join(sorted({h[1] for h in uniq}))

    return 1, observed_values, matched_tech_info, matched_cust_cols

def build_exception_flag(flag):
    # Customer phone matching technician phone is SUSPICIOUS → exception = yes
    return "yes" if flag == 1 else "no"


def build_expected_value():
    return ("Customer contact numbers (calling_from_number, customer_contact, "
            "customer_registered_phone, customer_alternate_phone) should NOT match "
            "any registered technician phone number")


def build_observed_value(flag, matched_phones, matched_cust_cols):
    if flag == 1:
        return (f"Customer phone(s) {matched_phones} matched with technician "
                f"records (fields involved: {matched_cust_cols})")
    return "No customer contact number matched with any technician record"


def build_additional_remarks(flag, matched_tech_info):
    if flag == 1:
        return (f"Potential conflict — customer contact appears in technician "
                f"database. Matched technician details: {matched_tech_info}")
    return "Clean — no overlap between customer contacts and technician phones"
